LOG_INTERVAL = 1  # seconds
I2C_ADDRESS = 0x76

MAX_POINTS = 120
BASELINE_SAMPLES = 40  # samples to establish baseline

# ---------------- GAS DETECTION & AI CLASSIFICATION ----------------

def estimate_iaq(gas_ohms, humidity):
    """Calculate Indoor Air Quality score (0-500, lower is better)"""
    gas_score = min(max((gas_ohms / 100000) * 100, 0), 100)
    humidity_score = max(0, 100 - abs(humidity - 40) * 2)
    iaq = 500 - (gas_score * 2.5 + humidity_score * 2)
    return max(0, min(500, iaq))

def classify_gases(gas_ohms, temp_c, humidity):
    """
    AI-based gas classification using resistance patterns.
    BME688 gas resistance varies with different gas types:
    - High resistance (>100kΩ): Clean air
    - Medium (20-100kΩ): VOCs present
    - Low (<20kΩ): Strong reducing gases (H2, CO, ethanol)
    """
   
    # Establish baseline
    if len(gas_baseline) < BASELINE_SAMPLES:
        gas_baseline.append(gas_ohms)
        return {
            'voc_ppm': 0, 'vsc_ppb': 0, 'co2_ppm': 400,
            'h2_ppm': 0, 'co_ppm': 0, 'ethanol_ppm': 0
        }
   
    baseline = np.median(gas_baseline)
    resistance_ratio = gas_ohms / baseline if baseline > 0 else 1
   
    # Add to history for pattern analysis
    gas_history.append(gas_ohms)
   
    # Calculate gas estimates based on resistance drop and patterns
    estimates = {}
   
    # VOC (Volatile Organic Compounds) - General air quality
    # Resistance drops with VOC presence
    if resistance_ratio < 0.9:
        voc_factor = (1 - resistance_ratio) * 100
        estimates['voc_ppm'] = min(voc_factor * 0.5, 10.0)
    else:
        estimates['voc_ppm'] = 0
   
    # VSC (Volatile Sulfur Compounds) - Detectable at very low levels
    # Sharp, sudden drops indicate VSC
    if len(gas_history) >= 3:
        recent_change = (gas_history[-1] - gas_history[-3]) / gas_history[-3]
        if recent_change < -0.15:  # Rapid drop
            estimates['vsc_ppb'] = min(abs(recent_change) * 1000, 500)
        else:
            estimates['vsc_ppb'] = 0
    else:
        estimates['vsc_ppb'] = 0
   
    # CO2 estimation (ambient is ~400ppm)
    # Moderate resistance drop, correlates with humidity
    humidity_factor = max(0, (humidity - 40) / 60)
    if resistance_ratio < 0.95:
        estimates['co2_ppm'] = 400 + (1 - resistance_ratio) * 2000 + humidity_factor * 500
    else:
        estimates['co2_ppm'] = 400
   
    # H2 (Hydrogen) - Strong reducing gas, large resistance drop
    if resistance_ratio < 0.5:
        estimates['h2_ppm'] = (1 - resistance_ratio) * 100
    else:
        estimates['h2_ppm'] = 0
   
    # CO (Carbon Monoxide) - Dangerous, moderate to large drop
    # Temperature compensated
    temp_compensation = 1 + (temp_c - 25) * 0.01
    if resistance_ratio < 0.7:
        estimates['co_ppm'] = max(0, (1 - resistance_ratio) * 50 * temp_compensation)
    else:
        estimates['co_ppm'] = 0
   
    # Ethanol - Large resistance drop, similar to alcohols
    if resistance_ratio < 0.6:
        estimates['ethanol_ppm'] = (1 - resistance_ratio) * 200
    else:
        estimates['ethanol_ppm'] = 0
   
    return estimates

def assess_health_risk(gas_est):
    """
    Assess health risk based on detected gas levels
    Returns: (risk_level, risk_description, color)
    """
    risks = []
   
    # VOC thresholds (ppm)
    if gas_est['voc_ppm'] > 5.0:
        risks.append("HIGH VOC")
    elif gas_est['voc_ppm'] > 2.0:
        risks.append("MODERATE VOC")
   
    # CO thresholds (ppm) - VERY DANGEROUS
    if gas_est['co_ppm'] > 50:
        return ("CRITICAL", "DANGEROUS CO LEVELS - EVACUATE!", "#ff0000")
    elif gas_est['co_ppm'] > 9:
        return ("HIGH", "Elevated CO detected - Ventilate immediately", "#ff6600")
    elif gas_est['co_ppm'] > 3:
        risks.append("Low CO detected")
   
    # CO2 thresholds (ppm)
    if gas_est['co2_ppm'] > 2000:
        risks.append("HIGH CO2")
    elif gas_est['co2_ppm'] > 1000:
        risks.append("Elevated CO2")
   
    # H2 thresholds (ppm)
    if gas_est['h2_ppm'] > 10:
        risks.append("H2 detected")
   
    # VSC thresholds (ppb)
    if gas_est['vsc_ppb'] > 100:
        risks.append("Strong odors")
   
    # Overall assessment
    if len(risks) >= 3 or any("HIGH" in r for r in risks):
        return ("HIGH", " | ".join(risks), "#ff6600")
    elif len(risks) > 0:
        return ("MODERATE", " | ".join(risks), "#ffaa00")
    else:
        return ("GOOD", "Air quality good", "#00ff00")

# ---------------- LOGGER THREAD ----------------
def log_sensor():
    while True:
        try:
            temp_c = bme.temperature
            temp_f = temp_c * 9 / 5 + 32
            humidity = bme.humidity
            pressure = bme.pressure
            gas = bme.gas
            altitude = bme.altitude
           
            # Calculate IAQ and classify gases
            iaq = estimate_iaq(gas, humidity)
            gas_est = classify_gases(gas, temp_c, humidity)
            risk_level, risk_desc, risk_color = assess_health_risk(gas_est)
           
            now = datetime.now()
            ts = now.strftime("%H:%M:%S")

            latest.update({
                "time": ts,
                "temp_f": temp_f,
                "temp_c": temp_c,
                "humidity": humidity,
                "pressure": pressure,
                "gas": gas,
                "altitude": altitude,
                "iaq": iaq,
                "risk_level": risk_level,
                "risk_desc": risk_desc,
                "risk_color": risk_color
            })
           
            gas_estimates.update(gas_est)

            with open(CSV_FILE, "a", newline="") as f:
                csv.writer(f).writerow([
                    now.isoformat(),
                    f"{temp_c:.2f}",
                    f"{temp_f:.2f}",
                    f"{humidity:.2f}",
                    f"{pressure:.2f}",
                    gas,
                    f"{altitude:.2f}",
                    f"{iaq:.1f}",
                    f"{gas_est['voc_ppm']:.3f}",
                    f"{gas_est['vsc_ppb']:.1f}",
                    f"{gas_est['co2_ppm']:.1f}",
                    f"{gas_est['h2_ppm']:.3f}",
                    f"{gas_est['co_ppm']:.3f}",
                    f"{gas_est['ethanol_ppm']:.3f}",
                    risk_level
                ])

            time_data.append(ts)
            temp_f_data.append(temp_f)
            humidity_data.append(humidity)
            pressure_data.append(pressure)
            gas_data.append(gas / 1000)  # KΩ
            iaq_data.append(iaq)

            if len(time_data) > MAX_POINTS:
                time_data.pop(0)
                temp_f_data.pop(0)
                humidity_data.pop(0)
                pressure_data.pop(0)
                gas_data.pop(0)
                iaq_data.pop(0)

            time.sleep(LOG_INTERVAL)
        except RuntimeError:
            time.sleep(1)

# ---------------- UPDATE FUNCTIONS ----------------
def update_readout():
    # Environmental readings
    env_labels["time"].config(text=latest.get("time", "--"))
    env_labels["temp_f"].config(text=f"{latest.get('temp_f', 0):.2f}")
    env_labels["temp_c"].config(text=f"{latest.get('temp_c', 0):.2f}")
    env_labels["humidity"].config(text=f"{latest.get('humidity', 0):.2f}")
    env_labels["pressure"].config(text=f"{latest.get('pressure', 0):.2f}")
    env_labels["altitude"].config(text=f"{latest.get('altitude', 0):.2f}")
   
    # Gas readings
    gas_labels["gas_kohms"].config(text=f"{latest.get('gas', 0)/1000:.2f}")
    gas_labels["iaq"].config(text=f"{latest.get('iaq', 0):.1f}")
    gas_labels["voc"].config(text=f"{gas_estimates.get('voc_ppm', 0):.3f}")
    gas_labels["vsc"].config(text=f"{gas_estimates.get('vsc_ppb', 0):.1f}")
    gas_labels["co2"].config(text=f"{gas_estimates.get('co2_ppm', 0):.1f}")
    gas_labels["h2"].config(text=f"{gas_estimates.get('h2_ppm', 0):.3f}")
    gas_labels["co"].config(text=f"{gas_estimates.get('co_ppm', 0):.3f}")
    gas_labels["ethanol"].config(text=f"{gas_estimates.get('ethanol_ppm', 0):.3f}")
   
    # Color-code dangerous levels
    if gas_estimates.get('co_ppm', 0) > 9:
        gas_labels["co"].config(fg="#ff0000")
    elif gas_estimates.get('co_ppm', 0) > 3:
        gas_labels["co"].config(fg="#ffaa00")
    else:
        gas_labels["co"].config(fg="#58a6ff")
   
    if gas_estimates.get('voc_ppm', 0) > 5:
        gas_labels["voc"].config(fg="#ff6600")
    elif gas_estimates.get('voc_ppm', 0) > 2:
        gas_labels["voc"].config(fg="#ffaa00")
    else:
        gas_labels["voc"].config(fg="#58a6ff")
   
    if gas_estimates.get('co2_ppm', 0) > 2000:
        gas_labels["co2"].config(fg="#ff6600")
    elif gas_estimates.get('co2_ppm', 0) > 1000:
        gas_labels["co2"].config(fg="#ffaa00")
    else:
        gas_labels["co2"].config(fg="#58a6ff")
   
    # Health risk
    risk_level_label.config(
        text=latest.get('risk_level', 'INIT'),
        bg=latest.get('risk_color', '#555555')
    )
    risk_desc_label.config(text=latest.get('risk_desc', 'Collecting data...'))
   
    root.after(1000, update_readout)


def update_graphs():
    ax1.clear(); ax2.clear(); ax3.clear(); ax4.clear()

    # Temperature
    ax1.plot(time_data, temp_f_data, color="#58a6ff", linewidth=2)
    ax1.set_ylabel("Temp °F", color="#c9d1d9", fontsize=9)
    ax1.set_title("Temperature", color="#58a6ff", fontsize=10)

    # Humidity
    ax2.plot(time_data, humidity_data, color="#3fb950", linewidth=2)
    ax2.axhline(y=40, color='#6e7681', linestyle='--', linewidth=1, alpha=0.5)
    ax2.set_ylabel("Humidity %", color="#c9d1d9", fontsize=9)
    ax2.set_title("Humidity (optimal: ~40%)", color="#3fb950", fontsize=10)

    # Gas Resistance
    ax3.plot(time_data, gas_data, color="#d29922", linewidth=2)
    ax3.set_ylabel("Gas KΩ", color="#c9d1d9", fontsize=9)
    ax3.set_title("Gas Resistance (higher = cleaner air)", color="#d29922", fontsize=10)

    # IAQ Score
    colors = ['#3fb950' if val < 100 else '#d29922' if val < 200 else '#f85149' for val in iaq_data]
    ax4.scatter(range(len(iaq_data)), iaq_data, c=colors, s=20)
    ax4.plot(time_data, iaq_data, color="#8b949e", linewidth=1, alpha=0.5)
    ax4.axhline(y=100, color='#3fb950', linestyle='--', linewidth=1, alpha=0.3)
    ax4.axhline(y=200, color='#d29922', linestyle='--', linewidth=1, alpha=0.3)
    ax4.set_ylabel("IAQ", color="#c9d1d9", fontsize=9)
    ax4.set_xlabel("Time", color="#c9d1d9", fontsize=9)
    ax4.set_title("Indoor Air Quality (lower = better)", color="#f85149", fontsize=10)
   
    # Format x-axis for all plots
    for ax in (ax1, ax2, ax3, ax4):
        ax.set_facecolor("#161b22")
        ax.tick_params(colors="#8b949e", labelsize=7)
        ax.grid(True, alpha=0.2, color="#30363d")
        if len(time_data) > 0:
            step = max(1, len(time_data) // 6)
            ax.set_xticks(range(0, len(time_data), step))
            ax.set_xticklabels([time_data[i] if i < len(time_data) else ""
                                for i in range(0, len(time_data), step)],
                               rotation=45, ha='right')

    fig.tight_layout()
    canvas.draw()
    root.after(2000, update_graphs)

threading.Thread(target=log_sensor, daemon=True).start()
