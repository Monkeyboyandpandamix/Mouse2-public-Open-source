#!/usr/bin/env python3
"""
Servo Control Script for Raspberry Pi
Allows precise control of servo position with configurable min/max limits
"""

import time
import sys

try:
    import RPi.GPIO as GPIO
except RuntimeError as e:
    print("Error importing RPi.GPIO. This script must be run with sudo:")
    print("  sudo python3 Servo")
    sys.exit(1)

# Configuration
SERVO_PIN = 4         # GPIO  (Physical pin )
MIN_DUTY = 2.5         # Duty cycle for fully closed position (0 degrees)
MAX_DUTY = 12.5        # Duty cycle for fully open position (180 degrees)
PWM_FREQUENCY = 50     # Standard servo frequency (50Hz)

def setup_servo():
    """Initialize GPIO and PWM for servo control"""
    GPIO.setwarnings(False)
    GPIO.setmode(GPIO.BCM)
    GPIO.setup(SERVO_PIN, GPIO.OUT)
    pwm = GPIO.PWM(SERVO_PIN, PWM_FREQUENCY)
    pwm.start(0)
    return pwm

def set_angle(pwm, angle):
    """
    Set servo to specific angle
    angle: 0 (closed) to 180 (open)
    """
    if angle < 0:
        angle = 0
    elif angle > 180:
        angle = 180
    
    duty = MIN_DUTY + (angle / 180.0) * (MAX_DUTY - MIN_DUTY)
    pwm.ChangeDutyCycle(duty)
    time.sleep(0.3)  # Give servo time to move
    pwm.ChangeDutyCycle(0)  # Stop sending signal to prevent jitter

def close_servo(pwm):
    """Move servo to fully closed position (0 degrees)"""
    print("Closing servo...")
    set_angle(pwm, 0)

def open_servo(pwm):
    """Move servo to fully open position (180 degrees)"""
    print("Opening servo...")
    set_angle(pwm, 180)

def main():
    print("Initializing servo on GPIO 17 (Pin 11)...")
    
    try:
        pwm = setup_servo()
    except Exception as e:
        print(f"Error setting up GPIO: {e}")
        print("\nMake sure to run with sudo:")
        print("  sudo python3 Servo")
        sys.exit(1)
    
    try:
        print("Servo Control Started")
        print(f"GPIO Pin: {SERVO_PIN} (Physical Pin 11)")
        print(f"Min Position (closed): {MIN_DUTY}% duty cycle")
        print(f"Max Position (open): {MAX_DUTY}% duty cycle")
        print("\nCommands:")
        print("  'o' or 'open' - Open servo")
        print("  'c' or 'close' - Close servo")
        print("  'angle <0-180>' - Set specific angle")
        print("  'q' or 'quit' - Exit")
        
        while True:
            cmd = input("\nEnter command: ").strip().lower()
            
            if cmd in ['q', 'quit', 'exit']:
                break
            elif cmd in ['o', 'open']:
                open_servo(pwm)
            elif cmd in ['c', 'close']:
                close_servo(pwm)
            elif cmd.startswith('angle'):
                try:
                    angle = int(cmd.split()[1])
                    print(f"Setting angle to {angle} degrees...")
                    set_angle(pwm, angle)
                except (IndexError, ValueError):
                    print("Invalid angle. Use: angle <0-180>")
            else:
                print("Unknown command")
    
    except KeyboardInterrupt:
        print("\n\nInterrupted by user")
    
    finally:
        pwm.stop()
        GPIO.cleanup()
        print("GPIO cleaned up. Goodbye!")

if __name__ == "__main__":
    main()
